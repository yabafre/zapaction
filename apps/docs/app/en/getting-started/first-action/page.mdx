# First Action

```ts
import { defineAction } from "@zapaction/core";
import { z } from "zod";

export const listUsers = defineAction({
  input: z.object({}),
  output: z.array(z.object({ id: z.string(), name: z.string() })),
  tags: ["users"],
  handler: async () => [],
});
```

Then wire your first read hook with `useActionQuery` and a feature key.
