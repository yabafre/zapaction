# Getting Started

## Install

```bash
pnpm add @zapaction/core @zapaction/react @zapaction/query zod
```

## 1) Register server context

```ts
// app/bootstrap.server.ts
import "server-only";

import { setActionContext } from "@zapaction/core";

setActionContext(async () => ({
  tenantId: "demo-tenant",
}));
```

## 2) Register tag registry

```ts
// app/bootstrap.client.ts
"use client";

import { setTagRegistry } from "@zapaction/query";

setTagRegistry({
  users: [["users"]],
});
```

## 3) Build your first action

```ts
import { defineAction } from "@zapaction/core";
import { z } from "zod";

export const listUsers = defineAction({
  input: z.object({}),
  output: z.array(z.object({ id: z.string(), name: z.string() })),
  tags: ["users"],
  handler: async () => [],
});
```
